System Overview

The Assembly Quorum System is designed to manage and track attendance for assembly meetings, calculating whether quorum requirements are met based on shares/stock ownership.

1

System Purpose

Track member attendance and calculate quorum status for assembly meetings in real-time.

Feature Description
Member Management Track all assembly members with their share counts
Attendance Tracking Real-time check-in/check-out for meetings
Quorum Calculation Automatic calculation based on shares present
Real-time Updates Live dashboard showing quorum status
Reporting Generate attendance and quorum reports
2

Key Requirements

The system must handle specific quorum calculation rules.

Quorum Requirements
• Quorum calculated based on share ownership
• Minimum 25% of total shares must be present
• Members can represent multiple shares
• Proxy votes count toward quorum
• Real-time updates during meeting
• Historical tracking of all meetings

Quorum Calculation

1

Quorum Formula

The mathematical formula used to determine if quorum is met.

Quorum Calculation Formula
Quorum Percentage = (Σ Shares Present / Σ Total Shares) × 100
Where:
• Σ Shares Present = Sum of shares represented by present members
• Σ Total Shares = Total shares issued to all members
• Required Percentage = 25% (configurable)
• Quorum Met = Quorum Percentage ≥ Required Percentage

MySQL Schema

Normalized database design for assembly management.

Core Database Tables
-- Members Table CREATE TABLE members ( id INT PRIMARY KEY AUTO_INCREMENT, member_code VARCHAR(20) UNIQUE NOT NULL, full_name VARCHAR(100) NOT NULL, phone VARCHAR(20), email VARCHAR(100), shares_owned INT DEFAULT 0, member_type ENUM('regular', 'board', 'guest') DEFAULT 'regular', status ENUM('active', 'inactive', 'suspended') DEFAULT 'active', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Meetings Table CREATE TABLE meetings ( id INT PRIMARY KEY AUTO_INCREMENT, meeting_code VARCHAR(50) UNIQUE NOT NULL, title VARCHAR(200) NOT NULL, meeting_date DATETIME NOT NULL, quorum_required_percentage INT DEFAULT 25, total_shares_required INT AS (total_shares_issued * quorum_required_percentage / 100), status ENUM('scheduled', 'in_progress', 'completed', 'cancelled') DEFAULT 'scheduled', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Attendance Table CREATE TABLE attendance ( id INT PRIMARY KEY AUTO_INCREMENT, meeting_id INT NOT NULL, member_id INT NOT NULL, attendance_status ENUM('present', 'absent', 'proxy') DEFAULT 'absent', shares_present INT DEFAULT 0, check_in_time DATETIME, check_out_time DATETIME, FOREIGN KEY (meeting_id) REFERENCES meetings(id) ON DELETE CASCADE, FOREIGN KEY (member_id) REFERENCES members(id) ON DELETE CASCADE, UNIQUE KEY unique_attendance (meeting_id, member_id) ); -- Shares Table CREATE TABLE shares ( id INT PRIMARY KEY AUTO_INCREMENT, member_id INT NOT NULL, share_count INT NOT NULL, share_type VARCHAR(50), issued_date DATE, FOREIGN KEY (member_id) REFERENCES members(id) ON DELETE CASCADE );

PHP Backend Implementation

Quorum Calculation Class

PHP class for calculating quorum status.

<?php
class QuorumCalculator {
    private $db;
    
    public function __construct($db) {
        $this->db = $db;
    }
    
    // Calculate quorum for a specific meeting
    public function calculateQuorum($meetingId) {
        // Get total shares issued
        $totalSharesQuery = "SELECT SUM(shares_owned) as total_shares FROM members WHERE status = 'active'";
        $totalSharesResult = $this->db->query($totalSharesQuery);
        $totalShares = $totalSharesResult->fetch_assoc()['total_shares'] ?? 0;
        
        // Get shares present in meeting
        $presentSharesQuery = "SELECT SUM(a.shares_present) as present_shares 
            FROM attendance a 
            JOIN members m ON a.member_id = m.id 
            WHERE a.meeting_id = ? AND a.attendance_status IN ('present', 'proxy')";
        
        $stmt = $this->db->prepare($presentSharesQuery);
        $stmt->bind_param("i", $meetingId);
        $stmt->execute();
        $presentResult = $stmt->get_result();
        $presentShares = $presentResult->fetch_assoc()['present_shares'] ?? 0;
        
        // Calculate percentage
        $percentage = ($totalShares > 0) ? ($presentShares / $totalShares) * 100 : 0;
        
        // Get required percentage from meeting settings
        $meetingQuery = "SELECT quorum_required_percentage FROM meetings WHERE id = ?";
        $stmt = $this->db->prepare($meetingQuery);
        $stmt->bind_param("i", $meetingId);
        $stmt->execute();
        $meetingResult = $stmt->get_result();
        $meeting = $meetingResult->fetch_assoc();
        
        $requiredPercentage = $meeting['quorum_required_percentage'] ?? 25;
        $requiredShares = ($totalShares * $requiredPercentage) / 100;
        
        return [
            'total_shares' => $totalShares,
            'present_shares' => $presentShares,
            'percentage' => round($percentage, 2),
            'required_percentage' => $requiredPercentage,
            'required_shares' => ceil($requiredShares),
            'quorum_met' => $percentage >= $requiredPercentage,
            'remaining_shares' => max(0, ceil($requiredShares) - $presentShares)
        ];
    }
    
    // Update attendance status
    public function updateAttendance($meetingId, $memberId, $status, $sharesPresent) {
        $query = "INSERT INTO attendance (meeting_id, member_id, attendance_status, shares_present, check_in_time) 
                 VALUES (?, ?, ?, ?, NOW()) 
                 ON DUPLICATE KEY UPDATE 
                 attendance_status = VALUES(attendance_status), 
                 shares_present = VALUES(shares_present),
                 check_in_time = IF(VALUES(attendance_status) = 'present', NOW(), check_in_time),
                 check_out_time = IF(VALUES(attendance_status) = 'absent', NOW(), NULL)";
        
        $stmt = $this->db->prepare($query);
        $stmt->bind_param("iisi", $meetingId, $memberId, $status, $sharesPresent);
        
        return $stmt->execute();
    }
}
?>

Real-time Quorum Dashboard

JavaScript for live quorum updates.

class QuorumDashboard {
    constructor(apiUrl, meetingId) {
        this.apiUrl = apiUrl;
        this.meetingId = meetingId;
        this.pollingInterval = 5000; // 5 seconds
        this.init();
    }
    
    async init() {
        await this.loadQuorumData();
        this.startPolling();
        this.setupEventListeners();
    }
    
    async loadQuorumData() {
        try {
            const response = await fetch(`${this.apiUrl}/meeting/${this.meetingId}/quorum`);
            const data = await response.json();
            
            this.updateDashboard(data);
            this.updateProgressBar(data.percentage);
            this.updateMemberList(data.attendance);
            
        } catch (error) {
            console.error('Failed to load quorum data:', error);
        }
    }
    
    updateProgressBar(percentage) {
        const progressBar = document.getElementById('quorum-progress-bar');
        const percentageText = document.getElementById('quorum-percentage');
        
        if (progressBar) {
            progressBar.style.width = `${percentage}%`;
            progressBar.style.backgroundColor = percentage >= 25 ? 
                'var(--accent-green)' : 'var(--accent-red)';
        }
        
        if (percentageText) {
            percentageText.textContent = `${percentage.toFixed(1)}%`;
        }
    }
    
    updateDashboard(data) {
        document.getElementById('total-shares').textContent = 
            data.total_shares.toLocaleString();
        document.getElementById('present-shares').textContent = 
            data.present_shares.toLocaleString();
        document.getElementById('required-shares').textContent = 
            data.required_shares.toLocaleString();
        document.getElementById('quorum-status').textContent = 
            data.quorum_met ? 'QUORUM MET' : 'QUORUM NOT MET';
        document.getElementById('quorum-status').className = 
            data.quorum_met ? 'status-met' : 'status-not-met';
    }
    
    startPolling() {
        setInterval(() => {
            this.loadQuorumData();
        }, this.pollingInterval);
    }
    
    setupEventListeners() {
        // Handle member check-in/check-out
        document.querySelectorAll('.attendance-toggle').forEach(button => {
            button.addEventListener('click', (e) => {
                const memberId = e.target.dataset.memberId;
                const currentStatus = e.target.dataset.status;
                const newStatus = currentStatus === 'present' ? 'absent' : 'present';
                
                this.updateMemberAttendance(memberId, newStatus);
            });
        });
    }
    
    async updateMemberAttendance(memberId, status) {
        try {
            const response = await fetch(`${this.apiUrl}/attendance/update`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    meeting_id: this.meetingId,
                    member_id: memberId,
                    status: status
                })
            });
            
            if (response.ok) {
                await this.loadQuorumData();
            }
        } catch (error) {
            console.error('Failed to update attendance:', error);
        }
    }
}
